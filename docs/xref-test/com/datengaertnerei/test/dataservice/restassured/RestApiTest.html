29  @SpringBootTest(webEnvironment = WebEnvironment.RANDOM_PORT)
30  @Tag("rest")
31  class RestApiTest {
32  
33      @LocalServerPort
34      private int port;
35  
36  	@Autowired
37  	private PostalAddressRepository repository;
38  
39  	@BeforeEach
40  	public void before() {
41  		DataImportTest.ensureDataAvailability(repository);
42  	}
43      
44      @BeforeEach // BeforeAll runs ahead of local port injection 
45      void setup() {
46  		RestAssured.port = port;
47      }
48      
49  	@Test
50  	void shouldReturnCreditCard() {
51  		ValidatableResponse response = get("/api/v1/creditcard").then().assertThat().statusCode(200).
52  		        and().contentType(ContentType.JSON).and().body("number", notNullValue()).body("type", notNullValue())
53  				.body("cvc", notNullValue());
54  		String expiry = response.extract().jsonPath().getString("expiry");
55  		LocalDate localDate = LocalDate.parse(expiry, DateTimeFormatter.ISO_LOCAL_DATE);
56  		assertThat(localDate).isAfter(LocalDate.now());
57  	}
