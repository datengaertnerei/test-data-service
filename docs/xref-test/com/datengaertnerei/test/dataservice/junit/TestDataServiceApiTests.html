79  	 * Person/PostalAddress
80  	 */
81  	@Test
82  	void shouldReturnRandomPerson() {
83  
84  		Set<String> persons = new HashSet<>();
85  		SortedSet<PostalAddress> addresses = new TreeSet<>();
86  		// create a few more persons to hit more date of birth branches
87  		for (int i = 1; i < 10; i++) {
88  			assertPerson(persons, addresses, null);
89  			assertPerson(persons, addresses, AgeRange.ALL);
90  			assertPerson(persons, addresses, AgeRange.ADULT);
91  			assertPerson(persons, addresses, AgeRange.MINOR);
92  			assertPerson(persons, addresses, AgeRange.SENIOR);
93  		}
94  	}
95  
96  	private void assertPerson(Set<String> persons, SortedSet<PostalAddress> addresses, AgeRange range) {
97  		Person person = restController.person(range);
98  		assertThat(person).isNotNull();
99  		assertThat(person.getAddress()).isNotNull();
100 		String personString = stringifyPerson(person);
101 		assertThat(persons.contains(personString)).isFalse();
102 		addresses.add(person.getAddress());
103 		persons.add(personString);
104 
105 		if (null != range) {
106 			long yearsBetween = ChronoUnit.YEARS.between(person.getBirthDate(), LocalDate.now());
107 
108 			switch (range) {
109 			case ADULT:
110 				assertTrue(yearsBetween >= THRESHOLD_ADULT);
111 				break;
112 
113 			case MINOR:
114 				assertTrue(yearsBetween <= THRESHOLD_ADULT);
115 				break;
116 
117 			case SENIOR:
118 				assertTrue(yearsBetween >= THRESHOLD_SENIOR);
119 				break;
120 
121 			case ALL:
122 			default:
123 				break;
124 			}
125 		}
126 	}

276 		assertThat(result).isNotNull();

278 		assertThat(result.getBank().getBic()).hasSize(11);
279 		assertThat(result.getBank().getDesc()).isNotNull();
280 		assertThat(checkList.contains(result.getIban())).isFalse();
281 		try {

283 		} catch (IbanFormatException | InvalidCheckDigitException | UnsupportedCountryException e) {
284 			fail(e);

333 	@Test
334 	void shouldReturnOpenApi() {
335 
336 		try {
337 			//
338 			URL url = new URL("http://localhost:" + port + "/v3/api-docs");
339 			HttpURLConnection con = (HttpURLConnection) url.openConnection();
340 			int status = con.getResponseCode();
341 			if (HttpURLConnection.HTTP_OK != status) {
342 				fail("OpenAPI download failed with status " + status);
343 				con.disconnect();
344 			}
345 
346 			BufferedReader in = new BufferedReader(new InputStreamReader(con.getInputStream()));
347 			String inputLine;
348 			StringBuffer content = new StringBuffer();
349 			while ((inputLine = in.readLine()) != null) {
350 				content.append(inputLine);
351 			}
352 			in.close();
353 			con.disconnect();
354 
355 			JsonFactory factory = new JsonFactory();
356 			JsonParser parser = factory.createParser(content.toString());
357 			while (parser.nextToken() != null) {
358 			}
359 			parser.close();
360 
361 			BufferedWriter writer = new BufferedWriter(new FileWriter("target/test-data-service-oas.json"));
362 			writer.write(content.toString());
363 			writer.close();
364 
365 		} catch (IOException e) {
366 			fail(e);
367 		}
368 	}
