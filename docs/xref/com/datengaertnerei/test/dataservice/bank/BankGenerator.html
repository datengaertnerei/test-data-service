44  /**
45   * Spring Boot service to generate valid bank accounts
46   *
47   */
48  @Service
49  public class BankGenerator implements IBankGenerator {
50  	private static final String DEFAULT_CITY = "frankfurt am main";
51  
52  	private static Log log = LogFactory.getLog(BankGenerator.class);
53  
54  	private Map<String, Map<String, Bank>> bankDirectory;
55  	private Map<String, String> binList;
56  	private Random rnd;
57  
58  	/**
59  	 * ctor
60  	 */
61  	public BankGenerator() {
62  		rnd = new Random();
63  		rnd.setSeed(System.currentTimeMillis());
64  
65  		initBankDirectory();
66  		initBinList();
67  	}
68  
69  	private void initBinList() {
70  		binList = new HashMap<>();
71  		InputStream inStream = getClass().getResourceAsStream("binlist.csv");
72  		Reader in = new InputStreamReader(inStream);
73  
74  		Iterable<CSVRecord> records;
75  		try {
76  			records = CSVFormat.Builder.create().setDelimiter(';').build().parse(in);
77  			for (CSVRecord r : records) {
78  				if (r.size() >= 2) {
79  					String type = r.get(0);
80  					String bin = r.get(1);
81  					binList.put(bin, type);
82  				}
83  			}
84  		} catch (IOException e) {
85  			log.fatal("could not read BIN list", e);
86  		}
87  
88  	}
89  
90  	private void initBankDirectory() {
91  		bankDirectory = new HashMap<>();
92  		Map<String, String> bics = new HashMap<>();
93  		try {
94  			// get german bank codes from
95  			// https://www.bundesbank.de/de/aufgaben/unbarer-zahlungsverkehr/serviceangebot/bankleitzahlen/
96  			// and just use entries with checksum routine 00
97  			InputStream inStream = getClass().getResourceAsStream("banklist.csv");
98  			Reader in = new InputStreamReader(inStream);
99  
100 			Iterable<CSVRecord> records = CSVFormat.Builder.create().setDelimiter(';').setSkipHeaderRecord(true).build()
101 					.parse(in);
102 			for (CSVRecord r : records) {
103 				if (r.size() >= 5) {
104 					String bankCode = r.get(0);
105 					String desc = r.get(3);
106 					String city = r.get(2);
107 					String bic = r.get(4);
108 
109 					if (null == bic || 0 == bic.length()) {
110 						bic = bics.get(bankCode);
111 					} else {
112 						bics.put(bankCode, bic);
113 					}
114 
115 					Bank b = new Bank(bankCode, desc, bic, city);
116 					if (bankDirectory.containsKey(city.toLowerCase())) {
117 						Map<String, Bank> cityMap = bankDirectory.get(city.toLowerCase());
118 						cityMap.put(bankCode, b);
119 					} else {
120 						Map<String, Bank> cityMap = new HashMap<>();
121 						cityMap.put(bankCode, b);
122 						bankDirectory.put(city.toLowerCase(), cityMap);
123 					}
124 				}
125 			}
126 		} catch (IOException e) {
127 			log.fatal("could not read bank list", e);
128 		}
129 	}

131 	/**
132 	 * acct generation
133 	 */
134 	@Override
135 	public BankAccount generateAccount(String city) {
136 		// create valid internal account number
137 		Integer num = rnd.nextInt(999999999);
138 		int[] toCalc = Integer.toString(num).chars().map(c -> c - '0').toArray();
139 		Integer crc = calculateCheckDigit(toCalc);
140 
141 		// and convert to string
142 		String rawAccount = num.toString() + crc.toString();
143 		String account = "0000000000".substring(rawAccount.length()) + rawAccount;
144 
145 		// get banks for city or default
146 		Map<String, Bank> cityMap = bankDirectory.getOrDefault(city.toLowerCase(), bankDirectory.get(DEFAULT_CITY));
147 
148 		// fetch random bank from list
149 		int bankIndex = rnd.nextInt(cityMap.keySet().size());
150 		Optional<Bank> bankContainer = cityMap.values().stream().skip(bankIndex).findFirst();
151 		if (bankContainer.isEmpty()) {
152 			BankAccount result = new BankAccount();
153 			result.setComment("internal error while retrieving bank from list");
154 			return result;
155 		}
156 		Bank bank = bankContainer.get();
157 		// and build IBAN
158 		Iban iban = new Iban.Builder().countryCode(CountryCode.DE).bankCode(bank.getBankCode()).accountNumber(account)
159 				.build();
160 
161 		// combine to result
162 		BankAccount result = new BankAccount();
163 		result.setBank(bank);
164 		result.setIban(iban.toString());
165 		if (!result.getBank().getCity().equalsIgnoreCase(city)) {
166 			result.setComment("city not found in bank directory, switched to default");
167 		}
168 
169 		return result;
170 	}

172 	/**
173 	 * cc generation
174 	 */
175 	@Override
176 	public CreditCard generateCreditCard() {
177 		// fetch random bin from list
178 		int binIndex = rnd.nextInt(binList.keySet().size());
179 		Optional<String> bin = binList.keySet().stream().skip(binIndex).findFirst();
180 		if (bin.isEmpty()) {
181 			return new CreditCard(null, null, null, LocalDate.now());
182 		}
183 
184 		// create valid cc number
185 		String randomPart = Integer.toString(rnd.nextInt(999999999));
186 		String combined = bin.get() + "000000000".substring(randomPart.length()) + randomPart;
187 		int[] toCalc = combined.chars().map(c -> c - '0').toArray();
188 		Integer crc = calculateCheckDigit(toCalc);
189 
190 		// and convert to string
191 		String number = combined + crc.toString();
192 
193 		// CVC must be 3 digits
194 		int cvc = rnd.nextInt(899) + 100;
195 
196 		// create expiry date in the future, min. 6 months, max. 4 years
197 		LocalDate expiry = LocalDate.now().plusDays(rnd.nextInt(1250) + 210L);
198 
199 		// combine to result
200 		return new CreditCard(number, binList.get(bin.get()), Integer.toString(cvc), expiry);
201 	}
202 
203 	// calculate checksum for domestic account number, IBAN checksum is included in
204 	// IBAN builder
205 	private int calculateCheckDigit(int[] digits) {
206 
207 		/* double every other starting from right - jumping from 2 in 2 */
208 		for (int i = digits.length - 1; i >= 0; i -= 2) {
209 			digits[i] += digits[i];
210 
211 			/* taking the sum of digits grater than 10 - simple trick by substract 9 */
212 			if (digits[i] >= 10) {
213 				digits[i] = digits[i] - 9;
214 			}
215 		}
216 		int sum = 0;
217 		for (int i = 0; i < digits.length; i++) {
218 			sum += digits[i];
219 		}
220 		/* multiply by 9 step */
221 		sum = sum * 9;
222 
223 		return sum % 10;
224 	}
