44  /**
45   * Spring Boot service to generate valid bank accounts
46   *
47   */
48  @Service
49  public class BankGenerator implements IBankGenerator {
50  	private static final String DEFAULT_CITY = "frankfurt am main";
51  
52  	private static Log log = LogFactory.getLog(BankGenerator.class);
53  
54  	private Map<String, Map<String, Bank>> bankDirectory;
55  	private Map<String, String> binList;
56  	private Random rnd;
57  
58  	/**
59  	 * ctor
60  	 */
61  	public BankGenerator() {
62  		rnd = new Random();
63  		rnd.setSeed(System.currentTimeMillis());
64  
65  		initBankDirectory();
66  		initBinList();
67  	}
68  
69  	private void initBinList() {
70  		binList = new HashMap<>();
71  		InputStream inStream = getClass().getResourceAsStream("binlist.csv");
72  		Reader in = new InputStreamReader(inStream);
73  
74  		Iterable<CSVRecord> records;
75  		try {
76  			records = CSVFormat.Builder.create().setDelimiter(';').build().parse(in);
77  			for (CSVRecord r : records) {
78  				if (r.size() >= 2) {
79  					String type = r.get(0);
80  					String bin = r.get(1);
81  					binList.put(bin, type);
82  				}
83  			}
84  		} catch (IOException e) {
85  			log.fatal("could not read BIN list", e);
86  		}
87  
88  	}
89  
90  	private void initBankDirectory() {
91  		bankDirectory = new HashMap<>();
92  		try {
93  			// get german bank codes from
94  			// https://www.bundesbank.de/de/aufgaben/unbarer-zahlungsverkehr/serviceangebot/bankleitzahlen/
95  			// and just use entries with checksum routine 00
96  			InputStream inStream = getClass().getResourceAsStream("banklist.csv");
97  			Reader in = new InputStreamReader(inStream);
98  
99  			Iterable<CSVRecord> records = CSVFormat.Builder.create().setDelimiter(';').setSkipHeaderRecord(true).build()
100 					.parse(in);
101 			for (CSVRecord r : records) {
102 				if (r.size() >= 5) {
103 					String bankCode = r.get(0);
104 					String desc = r.get(3);
105 					String city = r.get(2);
106 					String bic = r.get(4);
107 
108 					Bank b = new Bank(bankCode, desc, bic, city);
109 					if (bankDirectory.containsKey(city.toLowerCase())) {
110 						Map<String, Bank> cityMap = bankDirectory.get(city.toLowerCase());
111 						cityMap.put(bankCode, b);
112 					} else {
113 						Map<String, Bank> cityMap = new HashMap<>();
114 						cityMap.put(bankCode, b);
115 						bankDirectory.put(city.toLowerCase(), cityMap);
116 					}
117 				}
118 			}
119 		} catch (IOException e) {
120 			log.fatal("could not read bank list", e);
121 		}
122 	}

124 	/**
125 	 * acct generation
126 	 */
127 	@Override
128 	public BankAccount generateAccount(String city) {
129 		// create valid internal account number
130 		Integer num = rnd.nextInt(999999999);
131 		int[] toCalc = Integer.toString(num).chars().map(c -> c - '0').toArray();
132 		Integer crc = calculateCheckDigit(toCalc);
133 
134 		// and convert to string
135 		String rawAccount = num.toString() + crc.toString();
136 		String account = "0000000000".substring(rawAccount.length()) + rawAccount;
137 
138 		// get banks for city or default
139 		Map<String, Bank> cityMap = bankDirectory.getOrDefault(city.toLowerCase(), bankDirectory.get(DEFAULT_CITY));
140 
141 		// fetch random bank from list
142 		int bankIndex = rnd.nextInt(cityMap.keySet().size());
143 		Optional<Bank> bankContainer = cityMap.values().stream().skip(bankIndex).findFirst();
144 		Bank bank = bankContainer.orElseThrow();
145 		// and build IBAN
146 		Iban iban = new Iban.Builder().countryCode(CountryCode.DE).bankCode(bank.getBankCode()).accountNumber(account)
147 				.build();
148 
149 		// combine to result
150 		BankAccount result = new BankAccount();
151 		result.setBank(bank);
152 		result.setIban(iban.toString());
153 		if (!result.getBank().getCity().equalsIgnoreCase(city)) {
154 			result.setComment("city not found in bank directory, switched to default");
155 		}
156 
157 		return result;
158 	}
159 
160 	/**
161 	 * cc generation
162 	 */
163 	@Override
164 	public CreditCard generateCreditCard() {
165 		// fetch random bin from list
166 		int binIndex = rnd.nextInt(binList.keySet().size());
167 		Optional<String> bin = binList.keySet().stream().skip(binIndex).findFirst();
168 
169 		// create valid cc number
170 		String randomPart = Integer.toString(rnd.nextInt(999999999));
171 		String combined = bin.orElseThrow() + "000000000".substring(randomPart.length()) + randomPart;
172 		int[] toCalc = combined.chars().map(c -> c - '0').toArray();
173 		Integer crc = calculateCheckDigit(toCalc);
174 
175 		// and convert to string
176 		String number = combined + crc.toString();
177 
178 		// CVC must be 3 digits
179 		int cvc = rnd.nextInt(899) + 100;
180 
181 		// create expiry date in the future, min. 6 months, max. 4 years
182 		LocalDate expiry = LocalDate.now().plusDays(rnd.nextInt(1250) + 210L);
183 
184 		// combine to result
185 		return new CreditCard(number, binList.get(bin.orElseThrow()), Integer.toString(cvc), expiry);
186 	}
187 
188 	// calculate checksum for domestic account number, IBAN checksum is included in
189 	// IBAN builder
190 	private int calculateCheckDigit(int[] digits) {
191 
192 		/* double every other starting from right - jumping from 2 in 2 */
193 		for (int i = digits.length - 1; i >= 0; i -= 2) {
194 			digits[i] += digits[i];
195 
196 			/* taking the sum of digits grater than 10 - simple trick by substract 9 */
197 			if (digits[i] >= 10) {
198 				digits[i] = digits[i] - 9;
199 			}
200 		}
201 		int sum = 0;
202 		for (int i = 0; i < digits.length; i++) {
203 			sum += digits[i];
204 		}
205 		/* multiply by 9 step */
206 		sum = sum * 9;
207 
208 		return sum % 10;
209 	}
