24  package com.datengaertnerei.test.dataservice;
25  
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  import org.springframework.beans.factory.annotation.Value;
29  import org.springframework.boot.CommandLineRunner;
30  import org.springframework.boot.SpringApplication;
31  import org.springframework.boot.autoconfigure.SpringBootApplication;
32  import org.springframework.boot.autoconfigure.security.servlet.SecurityAutoConfiguration;
33  import org.springframework.context.ApplicationContext;
34  import org.springframework.context.ConfigurableApplicationContext;
35  import org.springframework.context.annotation.Bean;
36  
37  import io.swagger.v3.oas.models.ExternalDocumentation;
38  import io.swagger.v3.oas.models.OpenAPI;
39  import io.swagger.v3.oas.models.info.Info;
40  import io.swagger.v3.oas.models.info.License;
41  
42  /**
43   * Spring Boot main application for test data service
44   *
45   */
46  @SpringBootApplication(exclude = {SecurityAutoConfiguration.class })
47  public class DataServiceApplication {
48  	private static final String IMPORT_ONLY_YES = "YES";
49  	private static Logger log = LoggerFactory.getLogger(DataServiceApplication.class);
50  
51  	@Value("${application.version}")
52  	private String apiVersion;
53  	
54  	/**
55  	 * @param args cli arguments
56  	 */
57  	public static void main(String[] args) {
58  		SpringApplication.run(DataServiceApplication.class, args);
59  	}
60  
61  	/**
62  	 * @param ctx Spring Boot app contect
63  	 * @return Spring Boot runner object
64  	 */
65  	@Bean
66  	public CommandLineRunner commandLineRunner(ApplicationContext ctx) {
67  		return args -> {
68  
69  			String[] beanNames = ctx.getBeanDefinitionNames();
70  			log.info("Started with bean count of {}", beanNames.length);
71  
72  			String importOnly = System.getenv("OSM_IMPORT_ONLY");
73  			if (null != importOnly && IMPORT_ONLY_YES.equalsIgnoreCase(importOnly)) {
74  				log.info("Import only mode - exiting");
75  				((ConfigurableApplicationContext) ctx).close();
76  			}
77  
78  		};
79  	}
80  
81  	/**
82  	 * @return OpenAPI doc bean
83  	 */
84  	@Bean
85  	public OpenAPI springOpenAPI() {
86  		return new OpenAPI()
87  				.info(new Info().title("Datengärtnerei Test Data Service API")
88  						.description("We provide ad hoc generated test data for fictitious german persons.")
89  						.version(apiVersion)
90  						.license(new License().name("MIT License")
91  								.url("https://github.com/datengaertnerei/test-data-service/blob/master/LICENSE")))
92  				.externalDocs(
93  						new ExternalDocumentation().description("Datengärtnerei").url("https://datengaertnerei.com/"));
94  	}
95  }
