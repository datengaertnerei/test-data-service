83  	/**
84  	 * 
85  	 */
86  	@PostConstruct
87  	public void init() {
88  		repoCount = 0L;
89  		try {
90  			surnames = loadValues(RES_SURNAMES);
91  			femaleNames = loadValues(RES_FEMALE);
92  			maleNames = loadValues(RES_MALE);
93  			eyecolors = loadValues(RES_EYECOLORS);
94  			professions = loadValues(RES_PROFESSIONS);
95  		} catch (IOException e) {
96  			log.error("Could not init value list:", e);
97  		}
98  
99  		random = new Random(System.currentTimeMillis());
100 		Long minAddressId = null == repository.min() ? 0L : repository.min();
101 		Long maxAddressId = null == repository.max() ? 0L : repository.max();
102 
103 		String importFile = System.getenv("OSM_IMPORT_FILE");
104 		if(0L == maxAddressId && null == importFile) {
105 			importFile = defaultFile;
106 		}
107 		
108 		try {
109 			if (null != importFile && importFile.length() != 0) {
110 				log.info("Importing OSM dump");
111 				repository.deleteAll(); // does not reset JPA ID generator
112 				OsmPbfAddressImportUtil.importAddresses(importFile, repository);
113 				minAddressId = repository.min();
114 				maxAddressId = repository.max();
115 			}
116 			offset = minAddressId - 1L; // needed for random access by ID
117 		} catch (MalformedURLException e) {
118 			log.error("Could not import OSM data:", e);
119 		}
120 
121 		// address record IDs should be without gap to avoid errors
122 		if (getCount() < (maxAddressId - minAddressId)) {
123 			log.info("Postal address database contains gaps. Please reimport OSM data.");
124 		}
125 		log.info("Address count: " + getCount());
126 
127 		taxIdGenerator = new TaxIdGenerator(random);
128 	}
129 
130 	private Long getCount() {
131 		// cache address repo count, this method is called for every random address to return
132 		if(repoCount == 0) {
133 			repoCount = repository.count() < Integer.MAX_VALUE ? repository.count() : Integer.MAX_VALUE;
134 		}
135 		return repoCount;
136 	}

138 	/**
139 	 * Creates a single random person object with linked address.
140 	 *
141 	 * @return the new person object
142 	 */
143 	@Override
144 	public Person createRandomPerson(AgeRange range) {
145 		Person randomPerson = createBasicPerson(range);
146 		Optional<PostalAddress> address = repository.findById(randomId());
147 		if (address.isPresent()) {
148 			randomPerson.setAddress(address.get());
149 		}
150 
151 		return randomPerson;
152 	}

154 	/**
155 	 * Creates a single random person object with linked address inside a given
156 	 * city.
157 	 *
158 	 * @param city the city to fetch random address from
159 	 *
160 	 * @return the new person object
161 	 */
162 	@Override
163 	public Person createRandomPersonInCity(String city, AgeRange range) {
164 		Person randomPerson = createBasicPerson(range);
165 		List<PostalAddress> addresses = repository.findByAddressLocalityIgnoreCase(city);
166 		return compileResult(randomPerson, addresses);
167 	}

169 	/**
170 	 * Creates a single random person object with linked address inside a given
171 	 * postal code area.
172 	 *
173 	 * @param postalCode the postal code marking the area to fetch random address
174 	 *                   from
175 	 *
176 	 * @return the new person object
177 	 */
178 	@Override
179 	public Person createRandomPersonInArea(String postalCode, AgeRange range) {
180 		Person randomPerson = createBasicPerson(range);
181 		List<PostalAddress> addresses = repository.findByPostalCodeStartsWith(postalCode);
182 		return compileResult(randomPerson, addresses);
183 	}

185 	private Person compileResult(Person randomPerson, List<PostalAddress> addresses) {
186 		PostalAddress address = null;
187 		if (null != addresses && !addresses.isEmpty()) {
188 			address = addresses.get(random.nextInt(addresses.size()));
189 		} else {
190 			Optional<PostalAddress> addressContainer = repository.findById(randomId());
191 			if (addressContainer.isPresent()) {
192 				address = addressContainer.get();
193 				randomPerson.setComment("postal code not found in address base");
194 			} else {
195 				randomPerson.setComment("internal error while retrieving address");
196 			}
197 		}
198 
199 		randomPerson.setAddress(address);
200 
201 		return randomPerson;
202 	}

204 	/**
205 	 * Creates a valid and (most probably) unique email address at a test domain.
206 	 * Since the top level domain .test is reserved, these email addresses will
207 	 * never be routed to any real world email server. You have to configure your
208 	 * test email server for this.
209 	 *
210 	 * @param firstname   the given name of the person
211 	 * @param surname     the family name of the person
212 	 * @param dateOfBirth the date of birth of the person (to avoid duplicates)
213 	 * @return the new email address as string
214 	 */
215 	private String createEmailAddress(String firstname, String surname, LocalDate dateOfBirth) {
216 		StringBuilder email = new StringBuilder().append(firstname).append(surname).append(dateOfBirth.getYear())
217 				.append(EMAIL_TEST);
218 
219 		return Normalizer.normalize(email.toString(), Form.NFD).replaceAll("[^\\p{ASCII}]", "");
220 	}

222 	/**
223 	 * Creates a reasonable human height.
224 	 *
225 	 * @param gender male or female (add diverse if you like)
226 	 * @return the new height as int
227 	 */
228 	private int createRandomHeight(String gender) {
229 		double height;
230 		do {
231 			height = random.nextGaussian() * 10.0 + 165.0;
232 		} while (height < 150.0 || height > 190.0);
233 		if (MALE.equals(gender)) {
234 			height += 10.0;
235 		}
236 		return (int) Math.round(height);
237 	}

239 	/**
240 	 * Creates a valid and reasonable human date of birth.
241 	 *
242 	 * @return the date of birth
243 	 */
244 	private LocalDate createRandomDateOfBirth(AgeRange range) {
245 		double age;
246 		age = createAge(range);
247 
248 		int month = random.nextInt(11) + 1;
249 		int day = 0;
250 		switch (month) {
251 		case 1:
252 		case 3:
253 		case 5:
254 		case 7:
255 		case 8:
256 		case 10:
257 		case 12:
258 			day = random.nextInt(30) + 1;
259 			break;
260 		case 2:
261 			day = random.nextInt(27) + 1;
262 			break;
263 		case 4:
264 		case 6:
265 		case 9:
266 		case 11:
267 			day = random.nextInt(29) + 1;
268 			break;
269 		default: // there is no other month
270 			break;
271 		}
272 
273 		return LocalDate.of(LocalDate.now().getYear() - (int) Math.round(age), month, day);
274 	}

276 	private double createAge(AgeRange range) {
277 		double age;
278 		double min = 1.0; // you have to be born
279 		double max = 100.0; // a reasonable limit
280 
281 		switch (range) {
282 		case ADULT:
283 			min = 19.0; // tolerance due to random day and month
284 			max = 100.0;
285 			break;
286 
287 		case MINOR:
288 			min = 1.0;
289 			max = 17.0; // tolerance due to random day and month
290 			break;
291 
292 		case SENIOR:
293 			min = 66.0; // tolerance due to random day and month
294 			max = 100.0;
295 			break;
296 
297 		case ALL:
298 		default:
299 			break;
300 		}
301 
302 		// choose factor smaller than medium to tighten curve
303 		double shiftFactor = (max - min) / 3;
304 		double mediumAge = min + (max - min) / 2;
305 		age = random.nextGaussian() * shiftFactor + mediumAge;
306 		if (age < min || age > max) {
307 			// cut off long tails
308 			age = mediumAge;
309 		}
310 		return age;
311 	}

313 	/**
314 	 * @return
315 	 */
316 	private Long randomId() {
317 		Long result = (long) random.nextInt(getCount().intValue());
318 
319 		// add offset for first ID
320 		return result + offset;
321 	}

323 	/**
324 	 * @param age
325 	 * @return
326 	 */
327 	private Person createBasicPerson(AgeRange range) {
328 		String gender = random.nextBoolean() ? MALE : FEMALE; // (add diverse if you like)
329 		String firstname;
330 		if (MALE.equals(gender)) {
331 			firstname = maleNames.get(random.nextInt(maleNames.size())).trim();
332 		} else {
333 			firstname = femaleNames.get(random.nextInt(femaleNames.size())).trim();
334 		}
335 		String surname = surnames.get(random.nextInt(surnames.size())).trim();
336 
337 		String eyecolor = eyecolors.get(random.nextInt(eyecolors.size())).trim();
338 
339 		LocalDate dateOfBirth = createRandomDateOfBirth(range);
340 		String emailAddress = createEmailAddress(firstname, surname, dateOfBirth);
341 		int height = createRandomHeight(gender);
342 
343 		String taxId = taxIdGenerator.createTaxId();
344 
345 		Person randomPerson = new Person(firstname, surname, gender, dateOfBirth, height, eyecolor, emailAddress,
346 				taxId);
347 		randomPerson.setProfession(professions.get(random.nextInt(professions.size())).trim());
348 
349 		if (1 == random.nextInt(5)) {
350 			String birthName = surnames.get(random.nextInt(surnames.size())).trim();
351 			randomPerson.setBirthName(birthName);
352 		}
353 		return randomPerson;
354 	}

356 	/**
357 	 * Loads value list with classloader for human-readable random person
358 	 * attributes.
359 	 *
360 	 * @param fileName name of the files
361 	 * @return list of values
362 	 * @throws IOException values could not be loaded
363 	 */
364 	private List<String> loadValues(String fileName) throws IOException {
365 		InputStream input = getClass().getResourceAsStream(fileName);
366 		BufferedReader reader = new BufferedReader(new InputStreamReader(input, StandardCharsets.UTF_8));
367 		ArrayList<String> values = new ArrayList<>();
368 		String line;
369 		while ((line = reader.readLine()) != null) {
370 			values.add(line);
371 		}
372 		reader.close();
373 
374 		return values;
375 	}
